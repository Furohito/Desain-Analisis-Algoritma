PROSEDUR CARI_KPK(a, b)
    KELIPATAN <- maksimum(a, b)
    
    ULANGI
        JIKA KELIPATAN MOD a = 0 DAN KELIPATAN MOD b = 0 MAKA
            KPK <- KELIPATAN
            KELUAR DARI PERULANGAN
        AKHIR JIKA
        KELIPATAN <- KELIPATAN + 1
    SAMPAI KETEMU
    
    KEMBALIKAN KPK
AKHIR PROSEDUR
